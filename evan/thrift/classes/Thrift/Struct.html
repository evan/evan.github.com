257:       case field[:type]
258:       when Types::MAP

265:       when Types::LIST

267:         value.each do |elem|

269:         end
270:         oprot.write_list_end
271:       when Types::SET
272:         oprot.write_set_begin(field[:element][:type], value.size)

274:           write_data(oprot, v, field[:element])
